#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n;
    cin >> n;

    if (n % 4 != 0) {
        cout << "NO\n";
        return;
    }

    cout << "YES\n";

    vector<int> v;

    // Even numbers
    for (int i = 1; i <= n / 2; i++) {
        v.push_back(2 * i);
    }

    // First (n/2 - 1) odd numbers
    for (int i = 1; i < n / 2; i++) {
        v.push_back(2 * i - 1);
    }

    // Last element
    v.push_back(3 * (n / 2) - 1);

    for (int x : v)
        cout << x << " ";
    cout << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;

    while (t--)
        solve();

    return 0;
}
