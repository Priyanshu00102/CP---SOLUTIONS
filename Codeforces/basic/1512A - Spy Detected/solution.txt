#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n; cin>>n;
    vector<int>v(n);
    for(int &i:v){
        cin>>i;
    }
    map<int,int>mpp;
    for(int i=0;i<n; i++){
        mpp[v[i]]++;
    }
    int un=0;
    for(auto it:mpp){
        if(it.second==1) un=it.first;
    }
    for(int i=0; i<v.size(); ++i){
        if(v[i]==un){
            cout<<i+1<<'\n';
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
