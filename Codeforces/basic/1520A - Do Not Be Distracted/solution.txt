#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n; cin>>n;
    string s; cin>>s;
    for(char c='A'; c<='Z'; c++){
        int fst=n; int last=-1;
        for(int i=0; i<n; i++){
           if(s[i]==c){
            fst=min(fst,i);
            last=max(last,i);
           }
        }
         for (int i = fst; i <= last; i++) {
            if (s[i] != c) {
                cout << "NO\n";
                return;
            }
        }
    }
    cout << "YES\n";
    }
    


int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
