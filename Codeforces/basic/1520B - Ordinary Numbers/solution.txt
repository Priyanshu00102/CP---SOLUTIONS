#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n; cin>>n;
    if(n<10) cout<<n<<'\n';
    else{
        int cnt=0;
        for(int i=1; i<=9; i++){
            int x=0;
            for(int j=1; j<=9; j++){
                x=x*10+1;
                if(x*i<=n) cnt++;
            }
        }
        cout<<cnt<<'\n';
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
