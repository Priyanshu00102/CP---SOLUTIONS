#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n; cin>>n;
    int mini=INT_MAX;
    vector<int>v;
    for(int i=0; i<n; i++){
        int x; cin>>x;
        v.push_back(x);
       mini=min(x,mini);
    }
    int cnt=0;
    for(int i: v){
        if(mini==i) cnt++;
    }
    
     cout<<n-cnt<<'\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
