#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n; cin>>n;
    int sum=0;
    for(int i=0; i<n; i++){
        int x; cin>>x;
        sum+=x;
    }
    if(sum==n) cout<<0<<'\n';
    else if(sum>n) cout<<sum-n<<'\n';
    else cout<<1<<'\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
