#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n;cin>>n;
    vector<int>v(n);
    int sum=0;
    for(int &i: v){
        cin>>i;
        sum+=i;
    }
   
    if(sum%n!=0) cout<<-1<<'\n';
    else {
         int cnt=0;
        int avg=sum/n;
        for(int i=0; i<n; i++){
            if(avg<v[i]) cnt++;
        }
        cout<<cnt<<'\n';
    }

}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
