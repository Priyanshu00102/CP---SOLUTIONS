#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n; cin>>n;
    vector<int>v(n);
    for(int i = 0; i < n; i++){
    v[i] = i + 1;
    }
    if(n%2==0){
        for(int i=0; i<n-1; i+=2){
            swap(v[i],v[i+1]);
        }
    }
    else{
        v[0]=3;v[1]=1; v[2]=2;
        for(int i=3; i<n-1; i+=2){
            swap(v[i],v[i+1]);
        }
       
    }
    for(int x: v){
        cout<<x<<" ";
    }
    cout<<'\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
