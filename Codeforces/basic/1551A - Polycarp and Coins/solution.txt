#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    ll n; cin>>n; int a=0,b=0;
   if(n%3==0) {
    a=n/3; b=n/3;
   }else if(n%3==1){
    a=(n/3)+1; b=n/3;
   }else{
    a=n/3; b=(n/3)+1;
   }
   cout<<a<<" "<<b<<'\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
