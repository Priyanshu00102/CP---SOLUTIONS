#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    string s; cin>>s;
    map<char,int>mpp;
    for(int i=0; i<s.size(); i++){
        mpp[s[i]]++;
    }
    int c1=0, c2=0;
    for(auto i: mpp){
        if(i.second==1) c1++;
        else c2++;
    }
    cout<<(c1/2)+c2<<'\n';

}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
