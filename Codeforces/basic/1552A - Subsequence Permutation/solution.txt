#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n;cin>>n;
    string s; cin>>s;
    string g=s;
    sort(g.begin(),g.end());
    int cnt=0;
    for(int i=0; i<s.size(); ++i){
        if(s[i]!=g[i]) cnt++;
    }
    cout<<cnt<<'\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
