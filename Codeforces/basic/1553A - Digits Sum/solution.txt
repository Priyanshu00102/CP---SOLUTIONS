#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    ll n; cin>>n;
    if(n<9) cout<<0<<'\n';
    else cout<<(n+1)/10<<'\n';

}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
