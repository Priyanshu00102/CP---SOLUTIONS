#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n,k; cin>>n>>k;
    vector<int>v(n);
    for(int i=0; i<n; i++){
        cin>>v[i];
    }
    int cnt=0;
    for(auto it:v){
        if(v[k-1]<=it && it>0){
            cnt++;
        }
    }
    cout<<cnt;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--) solve();

    return 0;
}
