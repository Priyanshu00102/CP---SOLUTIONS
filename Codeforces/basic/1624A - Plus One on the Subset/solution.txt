#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n; cin>>n;
    vector<int>v(n);
    for(int &i: v){
        cin>>i;
    }
    int mini=INT_MAX; int maxi=INT_MIN;
    for(int i:v){
        mini=min(mini,i);
        maxi=max(maxi,i);
    }
    cout<<maxi-mini<<'\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
