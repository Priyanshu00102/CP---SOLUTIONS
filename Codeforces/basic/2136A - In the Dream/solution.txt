#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int a,b,c,d; cin>>a>>b>>c>>d;
    int mx1=max(a,b);
    int min1=min(a,b);
    c=c-a;d=d-b;
    int mx2=max(c,d);
    int min2=min(c,d);
    if(mx1<=2*(min1+1) && mx2<=2*(min2+1)){
        cout<<"YES\n";
    } else {
        cout<<"NO\n";
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
