#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int x,y; cin>>x>>y;
    if(x<y) cout<<"NO\n";
    else if(x==y) cout<<"YES\n";
    else {
        if(x%y==0){
            cout<<"YES\n";
        }else{
            cout<<"NO\n";
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
