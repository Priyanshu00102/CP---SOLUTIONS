#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int k; cin>>k; int arr[k];
    int maxi=-1, cnt2=0;
    for(int i=0; i<k; i++){
        cin>>arr[i];
        maxi=max(maxi,arr[i]);
        if(arr[i]==2){
            cnt2++;
        }
    }
    if(maxi>2 || cnt2>=2){
        cout<<"YES\n";
    }else{
        cout<<"NO\n";
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    cin >> t;
    while (t--) solve();

    return 0;
}
