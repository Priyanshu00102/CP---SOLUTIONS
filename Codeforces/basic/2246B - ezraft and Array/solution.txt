#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n;
    cin >> n;

    if (n == 1) {
        cout << 1 << "\n";
        return;
    }

    if (n == 2) {
        cout << -1 << "\n";
        return;
    }

    cout << "1 2 3 ";

    ll sum = 6;   // current sum of the constructed array

    for (int i = 4; i <= n; i++) {
        cout << sum << " ";
        sum *= 2;
    }

    cout << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;

    while (t--) {
        solve();
    }

    return 0;
}
