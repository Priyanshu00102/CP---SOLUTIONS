#include <bits/stdc++.h>
using namespace std;

void solve() {
    string s;
    cin >> s;

    int cnt = 0;

    for (int i = 0; i < s.size(); i++) {
        bool found = false;

        for (int j = 0; j < i; j++) {
            if (s[i] == s[j]) {
                found = true;
                break;
            }
        }

        if (!found)
            cnt++;
    }

    if (cnt % 2 == 0)
        cout << "CHAT WITH HER!";
    else
        cout << "IGNORE HIM!";
}

int main() {
    solve();
}
