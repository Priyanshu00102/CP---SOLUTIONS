#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n, x; cin>>n >>x;
    string s; cin>>s;
    while(x--){
        int i=0;
        while(i<s.size()-1){
            if(s[i]=='B'&&s[i+1]=='G'){
                swap(s[i],s[i+1]);
                i+=2;
            }else{
                i++;
            }
        }
    }
    cout<<s;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--) solve();

    return 0;
}
