#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    string s, ans=""; 
    cin>>s; 
    
    for(int i=0; i<s.size();i++){
        if(s[i]=='.') ans+='0';
        else if(s[i]=='-' && s[i+1]=='.') {ans+='1'; i++;}
        else if(s[i]=='-'&& s[i+1]=='-') {ans+='2'; i++;}
    }
    cout<<ans;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--) solve();

    return 0;
}
