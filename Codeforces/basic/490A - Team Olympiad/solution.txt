#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
   int n,x; cin>>n;
   vector<int>one,two,three;
   for(int i=1; i<=n; i++){
    cin>>x;
    if(x==1) one.push_back(i);
    else if(x==2) two.push_back(i);
    else three.push_back(i);
   }
   int team=min({one.size(),two.size(),three.size()});
   cout<<team<<'\n';
   for(int i=0; i<team; i++){
    cout<<one[i]<<" ";
    cout<<two[i]<<" ";
    cout<<three[i]<<" "<<'\n';
   }
    
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--) solve();

    return 0;
}
