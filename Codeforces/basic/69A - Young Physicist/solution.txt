#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n; cin>>n;
    ll sx=0, sy=0,sz=0;
    while(n--){
        ll x,y,z;
        cin>>x >>y >>z;
        sx+=x;
        sy+-y;
        sz+=z;

    }
    if(sx==0 && sy==0 && sz==0) cout<<"YES";
    else cout<<"NO";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--) solve();

    return 0;
}
