#include <bits/stdc++.h>
using namespace std;

#define ll long long

void solve() {
    int n; cin>>n;
    string s; cin>>s;
    unordered_map<char,int>mpp;
    for(auto it:s){
        mpp[it]++;
    }
    if(mpp['A']<mpp['D']) cout<<"Danik";
    else if(mpp['A']>mpp['D']) cout<<"Anton";
    else cout<<"Friendship";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--) solve();

    return 0;
}
