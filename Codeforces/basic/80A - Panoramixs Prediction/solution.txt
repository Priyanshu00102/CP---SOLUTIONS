#include <bits/stdc++.h>
using namespace std;

#define ll long long
bool prime(int n){
    for(int i=2; i<n; i++){
        if(n%i==0){
            return false;
        }
        }
        return true;
    }


void solve() {
    int a,b; cin>>a >>b;
    for(int i=a+1; ; ++i){
        if(prime(i)){
            if(i==b) cout<<"YES";
            else {
                cout<<"NO";
            }
             break;
        }
    }
    
    
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t = 1;
    // cin >> t;
    while (t--) solve();

    return 0;
}
